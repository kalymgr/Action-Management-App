def getSession(base, engine): 

""" 

Function that returns the databaseSession 

:param base: 

:param engine: 

:return: 

"""

base.metadata.bind = engine 

DBSession = sessionmaker(bind=engine) 

databaseSession = DBSession() 

return databaseSession
